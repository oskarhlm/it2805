  function isArmstrong(num) {
    const numStr = num.toString();

    let currSum = 0;
    for (let i = 0; i < numStr.length; i++) {
      const numAtIdx = Number(numStr[i]);
      currSum += Math.pow(numAtIdx, numStr.length);
    }

    return currSum === num;
  }
